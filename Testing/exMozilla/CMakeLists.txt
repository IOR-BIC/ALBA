PROJECT (exMozilla)

SET(PROJECT_SRCS 
  main.cpp
)

IF (WIN32)
  ADD_EXECUTABLE(${PROJECT_NAME} WIN32 ${PROJECT_SRCS} "${PROJECT_SOURCE_DIR}/app.rc")
  CONFIGURE_FILE( "${ALBA_SOURCE_DIR}/Gui/manifest.in" "${EXECUTABLE_OUTPUT_PATH}/Debug/${PROJECT_NAME}.exe.manifest" COPYONLY)
ELSE (WIN32)
  ADD_EXECUTABLE(${PROJECT_NAME} ${PROJECT_SRCS} )
ENDIF (WIN32)

SET(PROJECT_LIBS
  albaVME albaOperations
)
IF (ALBA_USE_WXMOZILLA)
  SET(PROJECT_LIBS
    ${PROJECT_LIBS}
    wxMozilla
    )
ENDIF (ALBA_USE_WXMOZILLA)

TARGET_LINK_LIBRARIES(${PROJECT_NAME} ${PROJECT_LIBS})
